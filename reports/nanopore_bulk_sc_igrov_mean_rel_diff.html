fill_missing_matrix <- function(x, all_rownames) {
    missing_rownames <- setdiff(all_rownames, rownames(x))
    missing_matrix <- matrix(
        0, nrow = length(missing_rownames), ncol = ncol(x)
    )
    rownames(missing_matrix) <- missing_rownames
    full_matrix <- rbind(x, missing_matrix)
    full_matrix <- full_matrix[all_rownames,]
    return(full_matrix)
}
mean_rel_diff <- function(x, y) {
    mean(abs(x - y) / ((x + y) / 2))
}
calculate_mean_rel_diff <- function(bulk_tpm, pseudobulk_tpm) {
    mean_rel_diff_values <- sapply(k_values, function(k_value) {
        mean_rel_diff(
            bulk_tpm,
            pseudobulk_tpm[names(bulk_tpm), as.character(k_value)]
        )
    })
    names(mean_rel_diff_values) = k_values
    return(mean_rel_diff_values)
}
top_n_mean_rel_diff <- function(bulk_tpm, pseudobulk_tpm, k_value) {
    bulk_tpm_values <- sort(bulk_tpm, decreasing = TRUE)
    pseudobulk_tpm_values <- pseudobulk_tpm[
        names(bulk_tpm_values), as.character(k_value)
    ]
    mean_rel_diff_values <- sapply(n_values, function(n) {
        if (length(bulk_tpm_values) < n) {
            return(NA)
        } else {
            return(mean_rel_diff(
                bulk_tpm_values[1:n], pseudobulk_tpm_values[1:n]
            ))
        }
    })
    return(mean_rel_diff_values)
}

plot_mean_rel_diff_top_n <- function(k_value) {
    plot_df <- list(
        data.frame(
            tool = "Isosceles",

                           each = length(n_values)),

                    isosceles_bulk_tpm_list[["Promethion"]],
                    isosceles_pseudobulk_tpm,
                    k_value
                ),
                top_n_mean_rel_diff(
                    isosceles_bulk_tpm_list[["MinION"]],
                    isosceles_pseudobulk_tpm,
                    k_value
                )
            )
        ),
        data.frame(
            tool = "IsoQuant",
            platform = rep(c("Promethion", "MinION"),
                           each = length(n_values)),
            n_value = rep(n_values, 2),
            mean_rel_diff = c(
                top_n_mean_rel_diff(
                    isoquant_bulk_tpm_list[["Promethion"]],
                    isoquant_pseudobulk_tpm,
                    k_value
                ),
                top_n_mean_rel_diff(
                    isoquant_bulk_tpm_list[["MinION"]],
                    isoquant_pseudobulk_tpm,
                    k_value
                )
            )
        ),
        data.frame(
            tool = "FLAMES",
            platform = rep(c("Promethion", "MinION"),
                           each = length(n_values)),
            n_value = rep(n_values, 2),
            mean_rel_diff = c(
                top_n_mean_rel_diff(
                    flames_bulk_tpm_list[["Promethion"]],
                    flames_pseudobulk_tpm,
                    k_value
                ),
                top_n_mean_rel_diff(
                    flames_bulk_tpm_list[["MinION"]],
                    flames_pseudobulk_tpm,
                    k_value
                )
            )
        ),
        data.frame(
            tool = "Sicelore",
            platform = rep(c("Promethion", "MinION"),
                           each = length(n_values)),
            n_value = rep(n_values, 2),
            mean_rel_diff = c(
                top_n_mean_rel_diff(
                    sicelore_bulk_tpm_list[["Promethion"]],
                    sicelore_pseudobulk_tpm,
                    k_value
                ),
                top_n_mean_rel_diff(
                    sicelore_bulk_tpm_list[["MinION"]],
                    sicelore_pseudobulk_tpm,
                    k_value
                )
            )
        )
    )
    plot_df <- do.call(rbind, plot_df)
    plot_df <- plot_df[!is.na(plot_df$mean_rel_diff),]
    ggplot(plot_df, mapping = aes(x = n_value,
                                  y = mean_rel_diff,
                                  col = tool,
                                  linetype = platform)) +
        geom_line(size = 1) +
        scale_color_manual(values = c(Isosceles = "black",
                                      IsoQuant = "gold",
                                      FLAMES = "tomato",
                                      Sicelore = "royalblue")) +
        scale_linetype_manual(values = c(Promethion = "solid",
                                         MinION = "dotted")) +
        scale_x_continuous(
            trans = "log10",
            n.breaks = 7
        ) +
        lims(y = c(0, 2)) +
        labs(
           x = "Top N transcripts",
           y = "Mean rel. diff.",
           col = "Tool",
           linetype = "Platform"
        ) +
        theme_bw() +
        theme(aspect.ratio = 1,
              legend.title = element_blank())
}
